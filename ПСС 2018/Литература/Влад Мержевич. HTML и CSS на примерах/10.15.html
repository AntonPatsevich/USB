Пример. Создание таблицы с помощью скрипта

<html>
<body>

<table width=400 border=1>
<script language="JavaScript">
for (i=1; i<6; i++) {
document.writeln("<tr>");
for (j=1; j<6; j++) document.write("<td>" + i + j + "</td>");
document.writeln("</tr>");
}
</script>
</table>

</body>
</html>
