CREATE TABLE PAY(
	ID INT IDENTITY(1,1) PRIMARY KEY,
	PRICE MONEY NOT NULL,
	PAYMENT_DATE DATE NOT NULL,
)
GO
CREATE TABLE TRANSPORT_TYPE(
	ID INT IDENTITY(1,1) PRIMARY KEY,
	TITLE NVARCHAR(20) NOT NULL
)
GO
CREATE TABLE TRANSPORT(
	ID INT IDENTITY(1,1) PRIMARY KEY,
	PRICE MONEY NOT NULL,
	ID_TRANSPORT_TYPE INT NOT NULL,
	CHECK (PRICE>0),
	FOREIGN KEY (ID_TRANSPORT_TYPE) REFERENCES TRANSPORT_TYPE(ID)   
) 
GO

CREATE TABLE BANK(
	ID INT IDENTITY(1,1) PRIMARY KEY,
	TITLE NVARCHAR(30) NOT NULL,
	ADDRES NVARCHAR(30) NOT NULL,
)
GO

CREATE TABLE BANK_ACCOUNT(
	ID INT IDENTITY(1,1) PRIMARY KEY,
	TITLE NVARCHAR(30) NOT NULL,
	BANK_ID INT NOT NULL,
	MONEY_COUNT MONEY NOT NULL,
	FOREIGN KEY (BANK_ID) REFERENCES BANK(ID)
	
)
CREATE TABLE LOADING_POINT(
	ID INT IDENTITY(1,1) PRIMARY KEY,
	ADDRES NVARCHAR(30) NOT NULL,
	PRICE MONEY DEFAULT (0), 
)
CREATE TABLE CONTRAGENT(
	ID INT IDENTITY(1,1) PRIMARY KEY,
	ADDRES NVARCHAR(30),
	PASSPORT_NUMBER INT,
	PHONE_NUMBER INT,
	PASSPORT_REG CHAR(2),
	TITLE NVARCHAR(60),
	BANK_ACCOUNT_ID INT,
	FOREIGN KEY (BANK_ACCOUNT_ID) REFERENCES BANK_ACCOUNT(ID),
	FOREIGN KEY (BANK_ACCOUNT_ID) REFERENCES BANK_ACCOUNT(ID),
	
)








GO
CREATE TABLE ORDERLIST (
	ID INT IDENTITY(1,1) PRIMARY KEY,
	PRICE_SUM MONEY NOT NULL,
	PAYMENT_DATE DATE,
	DATE_OF_EXECUTION DATE,
	DATE_OF_CONCLUSION DATE NOT NULL,
	PENALTY MONEY  DEFAULT (0),
	LOADING_POINT_ID INT,

		FOREIGN KEY (ID_TRANSPORT_TYPE) REFERENCES TRANSPORT_TYPE(ID),
		FOREIGN KEY (LOADING_POINT_ID) REFERENCES LOADING_POINT(ID),

)
