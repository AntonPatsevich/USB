
CREATE VIEW ORDERER_LIST
AS(
		SELECT 
	FULL_ORDER.ID,SAMMI_ORDER_ID,CUSTOMER_ID,ORDERER_ID,DATE_OF_CONCLUSION,LOADING_POINT_ID,
	TRANSPORT_ID,DRIVER_ID,PAY_ID,DATE_OF_EXECUTION,
	[PAY].[PAYMENT_DATE] as 'FINAL_PAY_DATE',
	[SAMMI_ORDER].[PAYMENT_DATE] as 'SAMMI_PAY_DATE',
	ADDRES,PASSPORT_NUMBER,PASSPORT_REG,
	PHONE_NUMBER,
	TITLE,BANK_ACCOUNT_ID
	FROM FULL_ORDER
		INNER JOIN SAMMI_ORDER ON [FULL_ORDER].[SAMMI_ORDER_ID] = [SAMMI_ORDER].[ID] 
		INNER JOIN PAY ON  [SAMMI_ORDER].[PAY_ID] = [PAY].[ID]
		INNER JOIN CONTRAGENT ON  [FULL_ORDER].[ORDERER_ID]=[CONTRAGENT].[ID]
)
